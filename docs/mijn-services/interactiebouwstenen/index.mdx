import Link from "@docusaurus/Link";
import {
  useCurrentSidebarSiblings,
  filterDocCardListItems,
  useDocById,
} from "@docusaurus/plugin-content-docs/client";
import styles from "@site/src/theme/DocCard/styles.module.css";

export const BouwsteenCard = ({ item }) => {
  const doc = useDocById(item.docId ?? undefined);
  const description = item.description ?? doc?.description;
  return (
    <article
      className="col col--6"
      style={{ marginBottom: "var(--ifm-spacing-vertical)" }}
    >
      <Link
        href={item.href}
        className={`card padding--lg ${styles.cardContainer}`}
      >
        <h3 className={styles.cardTitle}>
          {item.customProps?.icon && (
            <img
              src={item.customProps.icon}
              className={styles.cardIcon}
              style={{ objectFit: "contain" }}
              alt=""
              aria-hidden="true"
            />
          )}{" "}
          {item.label}
        </h3>
        {description && <p className={styles.cardDescription}>{description}</p>}
      </Link>
    </article>
  );
};

export const BouwsteenCardList = () => (
  <section className="row">
    {filterDocCardListItems(useCurrentSidebarSiblings()).map((item, i) => (
      <BouwsteenCard key={i} item={item} />
    ))}
  </section>
);

# Interactiebouwstenen

Een interactiebouwsteen beschrijft een herkenbaar onderdeel van publieke
dienstverlening vanuit gebruikersperspectief. De bouwsteen maakt duidelijk welke
behoefte wordt ondersteund, welke interacties daarbij horen en welke informatie
daarvoor betekenisvol is.

Vanuit de bouwsteen wordt de uitwerking verder geconcretiseerd in kanalen,
informatiemodellen, extensies op de Interactieservices API en aansluitprofielen.
Zo blijft dezelfde interactie bruikbaar in meerdere kanalen en met verschillende
bronnen.

Hieronder vindt u het overzicht van de bouwstenen die in deze documentatie zijn
uitgewerkt.
